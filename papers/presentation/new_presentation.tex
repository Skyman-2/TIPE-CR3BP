	\subsection{Résolution du système linéarisé}
	\begin{frame}
		\frametitle{Linéarisation des équations du mouvement}
		Les équations du mouvements sont
		\begin{equation*}
			\label{équations du mouvement}
			\begin{cases}
				\dot{x} = v_x
				\\
				\dot{y} = v_y
				\\
				\dot{v_x} = \ddot{x} = 2\omega\dot{y} - \frac{1}{m}\bar{U}_x
				\\
				\dot{v_y} = \ddot{y} = - 2\omega\dot{x} - \frac{1}{m}\bar{U}_y
			\end{cases}
		\end{equation*}
		Ce qui se linéarise en
		\begin{equation}
			\begin{bmatrix}
				0 & 0 & 1 & 0 \\
				0 & 0 & 0 & 1 \\
				-\frac{1}{m}\bar{U}_{xx}^i & -\frac{1}{m}\bar{U}_{xy}^i
				& 0 & 2\omega \\
				-\frac{1}{m}\bar{U}_{yx}^i & -\frac{1}{m}\bar{U}_{yy}^i
				& -2\omega & 0
			\end{bmatrix}
			=: A
		\end{equation}
		où $\bar{U}_{wz}^i = \frac{\partial ^2\bar{U}}{\partial w \partial z}(x_{L_i}, y_{L_i})$
	\end{frame}
	
	\begin{frame}
		\frametitle{Résolution}
		Le calcul du polynôme caractéristique $\chi_A$ de $A$ nous donne :
		\begin{equation*}
			A = P
			\begin{bmatrix}
				\lambda & 0 & 0 & 0 \\
				0 & -\lambda & 0 & 0 \\
				0 & 0 & i\nu & 0 \\
				0 & 0 & 0 & -i\nu
			\end{bmatrix}
			P^{-1}, \text{ avec } 
			{\footnotesize \begin{cases}
				\lambda = \omega\sqrt{1+2\sqrt{7}} \\
				\nu = \omega\sqrt{2\sqrt{7}-1}
			\end{cases}}
		\end{equation*}
		
		\pause
		On en déduit l'expression explicite des solutions dans le plan :
		{\scriptsize 		
		\begin{equation}
			\mathcal{S}_{lin}^{pos} = \{ t \to
			\begin{bmatrix}
				C_1e^{\lambda t} + C_2e^{-\lambda t} + C_3\cos(\nu t) + C_4 \sin(\nu t)
				\\
				-\sigma C_1e^{\lambda t} + \sigma C_2e^{-\lambda t} + \tau (C_3 \sin(\nu t) - C_4 \cos(\nu t))
			\end{bmatrix},
			(C_1,C_2,C_3,C_4) \in \mathbb{R}^4
			\}
		\end{equation}}
		
		où $\sigma > 0$ et $\tau < 0$ sont des constantes.
	\end{frame}

	\subsection{Introduction à la théorie de Floquet}
	
	\begin{frame}
		\frametitle{Une petite définition...}
		{\footnotesize
		\begin{mydefinition}{Matrice fondamentale}
			Soit $(y_1,\dots,y_n)$ un système fondamental de solutions d'une équation différentielle linéaire. Une matrice fondamentale de cette équation différentielle est l'application $t \mapsto \Phi(t)$ où $\Phi(t)$ est la matrice dont les colonnes sont les vecteurs $(y_1(t),\dots,y_n(t))$. Alors si le système est $X' = A(t)X$, $\Phi$ vérifie $\Phi' = A(t)\Phi$.
			
			On appelle \textit{matrice fondamentale principale} la matrice fondamentale qui vérifie que $\Phi(t_0) = I_n$.
		\end{mydefinition}
		
		\bigskip
		
		\textit{Remarque :}
		
		Si $y$ est une solution de l'équation différentielle, een notant $y_0 := y(t_0)$, alors $\forall t \in \mathbb{R}, y(t) = \Phi(t)y_0$ où $\Phi$ est la matrice fondamentale principale.
		}
	\end{frame}
	
	\begin{frame}
		\frametitle{...et deux autres...}
		{\footnotesize
		\begin{mydefinition}{Le flot}
			Soit un système non-linéaire $X' = f(X)$ où $f \in \mathcal{C}^1(\mathbb{R},\mathbb{R}^4)$. On définit le flot avec origine de temps $t_0$ comme l'application $ \eta :
			\begin{array}{rcl}
				\mathbb{R} \times \mathbb{R}^4 & \to & \mathbb{R}^4 \\
				(t,x_0) & \mapsto & x(t)
			\end{array}
			$ où $x$ correspond à l'unique solution au problème de Cauchy de condition initiale $x_0$ au temps $t_0$.
		\end{mydefinition}
		\bigskip
		\pause
		\begin{mydefinition}{Matrice de Monodromie}
			Soit une équation différentielle de la forme $X'=A(t)X$ où A est à coefficients $T$-périodiques. On appelle \textit{matrice de Monodromie} la matrice $M = \Phi(T)$ où $\Phi$ est la matrice fondamentale principale du système.
		\end{mydefinition}
		}
	\end{frame}
	
	\begin{frame}
		\frametitle{Sensibilité du système aux conditions initiales}
		{\small Soit $\gamma$ une solution $T$-périodique du système non-linéaire correspondant à la condition initiale $\gamma(t_0) = x_0$. Soit $\delta x_0 \in \mathbb{R}^4$ une petite perturbation initiale. On linéarise le flot le long de $\gamma$ :
		\begin{equation*}
			\eta(t,x_0+\delta x_0) = \eta(t,x_0) + D_{x_0}\eta(t,x_0)\delta x_0 + \underset{\delta x_0 \to 0}{o}(\delta x_0)
		\end{equation*}
		\pause		
		et on remarque qu'en notant $\Phi(t) := D_{x_0}\eta(t,x_0)$,
		\begin{equation*}
			\Phi ' = Df(\gamma(t))\Phi, \text{ et } \Phi(t_0) = I_4
		\end{equation*}
		\pause
		donc c'est la matrice fondamentale principale de $X' = Df(\gamma(t))X$. On en déduit le
		\begin{importantresult}{Lien matrice de Monodromie du système linéarisé/sensibilité aux C.I.}
			\begin{equation}
				\forall k \in \mathbb{N}^*, \delta x(kT) = M^k\delta x_0
			\end{equation}
		\end{importantresult}
		}
	\end{frame}

	\subsection*{Equations du mouvement et linéarisé}
	\begin{frame}
		\frametitle{Expression des équations du mouvement}{\small
		On connait $\bar{U}$ le potentiel apparent dans $R_{NG}$. Cela inclue les contributions de $F_{ie}$, $F_{1/M}$, et $F_{2/M}$. Il manque uniquement la force de Coriolis $F_{ic}$ qui n'est pas conservative donc n'apparait pas dans le potentiel. Or
		\begin{equation*}
			\vec{F}_{ic} = -2m\vec{\Omega} \times \vec{v} = -2m\Omega\vec{u}_z \times (\dot{x}\vec{u}_x + \dot{y}\vec{u}_y) 
		\end{equation*}
		
		Alors par principe fondamentale de la dynamique, on déduit que
		\begin{equation*}
			\begin{cases}
				\ddot{x} - 2\omega\dot{y} = - \frac{1}{m}\bar{U}_x \\
				\ddot{y} + 2\omega\dot{x} = - \frac{1}{m}\bar{U}_y
			\end{cases}
		\end{equation*}
		qui donne immédiatement les équations du mouvement.
		
		}
	\end{frame}
	
	\begin{frame}
		\frametitle{Linéarisation des équations du mouvement}{\small
		Avec une expansion en série de Taylor à l'ordre 1 en $(x_{Li}, y_{Li})$, on obtient
		\begin{equation*}
			\begin{cases}
				\bar{U}_x(x,y) = \bar{U}_x(x_i,y_i) + \frac{\partial \bar{U}_x}{\partial x} (x_i,y_i) \times (x - x_i) + \frac{\partial \bar{U}_x}{\partial y} (x_i,y_i) \times (y - y_i)
				\\
				\bar{U}_y(x,y) = \bar{U}_y(x_i,y_i) + \frac{\partial \bar{U}_y}{\partial x} (x_i,y_i) \times (x - x_i) + \frac{\partial \bar{U}_y}{\partial y} (x_i,y_i) \times (y - y_i)
			\end{cases}
		\end{equation*}
		donc en fixant l'origine spatiale, et du potentiel, en $Li$, le système linéarisé s'écrit
		\begin{equation*}
			\label{eq du mvt}
			\dot{X}
			=
			\begin{bmatrix}
				\dot{x} \\
				\dot{y} \\
				\dot{v_x} \\
				\dot{v_y}
			\end{bmatrix}
			=
			\underbrace{
				\begin{bmatrix}
					0 & 0 & 1 & 0 \\
					0 & 0 & 0 & 1 \\
					-\frac{1}{m}\bar{U}_{xx}^i & -\frac{1}{m}\bar{U}_{xy}^i
					& 0 & 2\omega \\
					-\frac{1}{m}\bar{U}_{yx}^i & -\frac{1}{m}\bar{U}_{yy}^i
					& -2\omega & 0
				\end{bmatrix}
			}_{=: A}
			\begin{bmatrix}
				x \\ y \\ v_x \\ v_y
			\end{bmatrix}
		\end{equation*}
		}
	\end{frame}
	
	
	\subsection*{Diagonalisation}
	
	
	\begin{frame}
		\frametitle{Calcul des valeurs et vecteurs propres}{\scriptsize
		\begin{columns}[T]
			\column{0.48\linewidth}
			Approximation des dérivées secondes des potentiels :
			\begin{equation*}
				\begin{cases}
					\bar{\mathcal{U}}_{xx} = -9m\omega^2
					\\
					\bar{\mathcal{U}}_{yy} = 3m\omega^2
					\\
					\bar{\mathcal{U}}_{xy} = \bar{\mathcal{U}}_{yx} = 0
				\end{cases}
			\end{equation*}
			qui nous permet de calculer $\chi_A$ le polynôme caractéristique de $A$. On a
			\begin{align*}
			\chi_A &= \begin{vmatrix}
				X & 0 & -1 & 0 
				\\
				0 & X & 0 & -1
				\\
				-9\omega^2 & 0 & X & - 2 \omega
				\\
				0 & 3\omega^2 & 2\omega & X
			\end{vmatrix} \\
			&= X^4 -2\omega^2X^2 - 27\omega^2 \\
			&= (X-\lambda)(X+\lambda)(X-i\nu)(X+i\nu)
			\end{align*}
			où $\begin{cases}
				\lambda := \omega\sqrt{1+2\sqrt{7}} \in \mathbb{R}_+ \\
				\nu := \omega\sqrt{2\sqrt{7}-1} \in \mathbb{R}_+
			\end{cases}$
			
			\column{0.02\textwidth}
			\centering
			\textcolor{gray!40}{\rule{0.2pt}{1\textheight}}
			
			\column{0.48\linewidth}
			Equation aux valeurs propres pour déterminer les vecteurs propres :
			\begin{equation*}
				\begin{cases}
					x_3 = \rho x_1
					\\
					x_4 = \rho x_2
					\\
					9\omega^2 x_1 + 2\omega x_4 = \rho x_3
					\\
					-3\omega^2 x_2 -2\omega x_3 = \rho x_4	
				\end{cases}
			\end{equation*}
			d'où
			\begin{equation*}
				\begin{cases}
					(9\omega^2-\rho^2) x_1 + 2\omega \rho x_2 = 0
					\\
					(-3\omega^2-\rho^2) x_2 -2\omega \rho x_1 =	0
				\end{cases}
			\end{equation*}
			donc $P = \begin{bmatrix}
					1 & 1 & 1 & 1 \\
					-\sigma & \sigma & -i\tau & i\tau \\
					\lambda & -\lambda & i\nu & -i\nu \\
					-\lambda\sigma & -\lambda\sigma & \nu\tau & \nu\tau
				\end{bmatrix}$, $\text{ avec } \sigma = \frac{9\omega^2-\lambda^2}{2\omega\lambda} \text{ et } \tau = -\frac{9\omega^2+\nu^2}{2\omega\nu}$
			
		\end{columns}
		}
	\end{frame}
	
	\subsection*{Existence d'une famille d'orbite à 1 paramètre}
	\begin{frame}[fragile]
		\frametitle{Orbites périodiques du système linéarisé}{\small
		Forme des solutions périodiques du système linéarisé :
		\begin{equation*}
			\mathcal{S}_p = \{ t \mapsto
			\begin{bmatrix}
			R\cos(\nu t - \varphi) \\ \tau RA\sin(\nu t - \varphi) \\ \nu R \sin(\nu t - \varphi) \\ \nu\tau R \cos(\nu t - \varphi)
			\end{bmatrix}, (R,\varphi) \in \mathbb{R}^2
			\}
		\end{equation*}
		et on choisit comme paramètre $R$ l'amplitude de l'orbite selon l'axe $Ox$. Sans perte de généralité, on peut aussi choisir un déphasage $\varphi = 0$. Cela impose
		\begin{equation*}
			\begin{cases}
				x(0) = R \\ y(0) = 0 \\ v_x(0) = 0 \\ v_y(0) = \nu\tau R
			\end{cases}
		\end{equation*}
		
		Grâce au module \verb|numpy|, on peut évaluer numériquement $A$ (cf. la partie Python des annexes). On obtient de cette manière que $\nu\tau = 2,37\cdot 10^{-5} \: s^{-1}$.
		}
	\end{frame}
	

	\subsection*{Théorie de Floquet}
	\begin{frame}
		\frametitle{Lien entre deux matrices fondamentales}
		
		{\small
		\textit{Démonstration :}
		Soit $y_0 \in \mathbb{R}^n$, il existe une unique solution $y$ de l'équation différentielle tel que $y(0) = y_0$. De plus,
		\begin{equation*}
			\forall t \in \mathbb{R}, y(t) = \Phi(t)\Phi(0)^{-1}y_0
		\end{equation*}
		et de même pour $\Psi$. Donc
		\begin{equation*}
			\forall t \in \mathbb{R}, \Phi(t)\Phi(0)^{-1}y_0 = \Psi(t)\Psi(0)^{-1}y_0
		\end{equation*}
		
		Etant vraie pour tout $y_0 \in \mathbb{R}^n$, on a
		\begin{equation*}
			\forall t \in \mathbb{R}, \Phi(t)\Phi(0)^{-1} = \Psi(t)\Psi(0)^{-1}
		\end{equation*}
		d'où le résultat en posant $M := \Psi(0)^{-1}\Phi(0)$. $M$ est en effet bien inversible car une matrice fondamentale est inversible pour tout $t \in \mathbb{R}$.
		}
	\end{frame}
	
	\begin{frame}
		\frametitle{Lien matrice de Monodromie/sensibilité aux C.I.}
		
		{\small
		On considère
		\begin{equation*}
			x'(t) = f(x(t)), \qquad f \in \mathcal{C}^2(\mathbb{R}^n,\mathbb{R}^n)
		\end{equation*}
		et on fixe $t_0$ l'origine du temps.
		
		Supposons que l'on ait une solution périodique $\gamma$ tel que
		\begin{equation*}
			\gamma(t+T) = \gamma(t), \qquad \gamma(t_0) = x_0
		\end{equation*}
		Soit $\delta x_0 \in \mathbb{R}^4$ une petite perturbation. On note $x(t) := \eta(x_0 + \delta x_0, t)$ la solution correspondante à la C.I. $x_0 + \delta x_0$. 
		\begin{columns}
			\column{0.4\linewidth}
			\begin{figure}
				\centering
				\includegraphics[width=1\linewidth]{../annexes/illustration_delta_x.png}
			\end{figure}
			
			\column{0.5\linewidth}
			On note $\xi(t) := x(t) - \gamma(t)$ l'écart à la solution périodique à un instant $t$. Dans l'équation différentielle, cela donne :
			\begin{equation*}
				\gamma'(t) + \xi'(t) = f(\gamma(t) + \xi(t))
			\end{equation*}
			
		\end{columns}
		}
		
		\photosources{Dynamical Systems, the Three-Body Problem and Space Mission Design}
	\end{frame}
	
	\begin{frame}
		
		{\small
		Par développement de Taylor,
		\begin{equation*}
			f(p + \xi) = f(p) + Df(p)\xi + o(\xi)
		\end{equation*}
		et en notant $\delta x$ l'approximation de $\xi$ au premier ordre,
		\begin{equation*}
			\delta x'(t) = Df(\gamma(t))\delta x(t)
		\end{equation*}
		car $\gamma$ vérifie $x'(t) = f(x(t))$.
		Comme $\gamma$ est $T$-périodique, $t \mapsto Df(\gamma(t)) =: A(t)$ l'est aussi.
		
		Or à $t$ fixé,
		\begin{equation*}
			\eta(t,x_0 + \delta x_0) = \eta(t,x_0) + D_{x_0}\eta(t,x_0)\delta x_0 + o(\delta x_0)
		\end{equation*}
		
		On pose $\Psi(t) := D_{x_0}\eta(t,x_0)$. Or par déf du flot, $\frac{\partial}{\partial t}\eta(t,x_0) = f(\eta(t,x_0))$. Alors en différentiant par rapport à $x_0$, et grâce au théorème de Schwartz ($f$ étant de classe $C^2$),
		\begin{equation*}
			\frac{d}{dt}\Psi(t) = Df(\gamma(t))\Psi(t)
		\end{equation*}
		et comme $\eta(t_0,x_0) = x_0$, $D_{x_0}\eta(t_0,x_0) = \Psi(t_0) = I_n$ donc inversible. $\Psi(t)$ est donc la matrice fondamentale principale du système.
		}
	\end{frame}
	
	
	\begin{frame}
		
		{\small
			
		\begin{importantresult}{Théorème de Floquet}
			Soit $X'=AX$ une équation différentielle où $A$ est à coefficients $T$-périodiques, et $\Phi$ la matrice fondamentale principale du système. Alors il existe $P \in \mathcal{C}^1_T(\mathbb{R}, \mathcal(M)_n(\mathbb{C}))$ et $B \in \mathcal(M)_n(\mathbb{C})$ tel que
			\begin{equation*}
				\forall t \in \mathbb{R}, \Phi(t) = P(t)e^{tB}
			\end{equation*}
			que l'on appelle la \textit{forme normale de Floquet}. En particulier, on appelle $e^{TB}=\Phi(T)$ la \textit{matrice de Monodromie} du système, et \textit{multiplicateurs caractéristiques du système} les valeurs propres de la matrice de Monodromie. Les multiplicateurs caractéristiques sont non nuls et indépendants du choix de $\Phi$ et du choix de $B$.
		\end{importantresult}
		
		Avec la forme normale de Floquet, on observe que
		\begin{equation*}
			M^k = (e^{TB})^k = e^{kTB} = P(kT)e^{kTB} = \Psi(kT)
		\end{equation*}
		et comme $\delta x(t) = \Psi(t)\delta x_0$, $\delta x(kT) = \Psi(kT)\delta x_0 = M^k \delta x_0$.
		}
	\end{frame}
	
	\begin{frame}
		\frametitle{Démonstration du théorème de Floquet}
		{\small
		Soit $t \in \mathbb{R}$. Par $T$-périodicité de $A$, on a 
		\begin{equation*}
			\begin{cases}
				\Phi'(t+T) = A(t+T)\Phi(t+T) = A(t)\Phi(t+T) \\
				\det(\Phi(t+T)) = \det(\Phi(t)) \neq 0
			\end{cases}
		\end{equation*}
		i.e. $t \mapsto \Phi(t+T)$ est une matrice fondamentale. Ainsi, il existe $M \in GL_n(\mathbb{C})$, tel que 
		\begin{equation*}
			\Phi(t+T) = \Phi(t)M
		\end{equation*}
		et par surjectivité de l'exponentielle sur $GL_n(\mathbb{C})$, il existe $B \in \mathcal{M}_n(\mathbb{C})$ tel que $M = e^{TB}$. 
		
		On pose $\forall t \in \mathbb{R}, P(t) := \Phi(t)e^{-tB}$. Pour tout $t \in \mathbb{R}$,
		\begin{equation*}
			P(t+T) = \Phi(t+T)e^{-(t+T)B} = \Phi(t)Me^{-TB}e^{-tB} = \Phi(t)e^{-tB} = P(t)
		\end{equation*}
		}
	\end{frame}
